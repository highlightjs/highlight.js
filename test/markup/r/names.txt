a1_foo, A1_FOO, .foo_, ._foo, Bar.42, foo..1, ., ._, .., ..., ..1, c, T, F, ._1

NA, NA_integer_, NA_real_, NA_character_, NA_complex_, NULL, NaN, Inf

function, while, repeat, for, if, in, else, next, break

return, switch, sum

# Non-primitive base functions

stop, try

`%*%`

`# x`
