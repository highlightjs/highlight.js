; macros
(defmacro unless [pred a b]
  `(if (not ~pred) ~a ~b))

; types
(deftype Circle [radius])
(deftype Square [length width])

;; multimethods again
(defmulti area class)
(defmethod area Circle [c]
    (* Math/PI (* (.radius c) (.radius c))))
(defmethod area Square [s]
    (* (.length s) (.width s)))
