<?php

/**
 * @param int $a
 * @return bool
 */
function isEven($a) {
    return ($a % 2) === 0;
}

/**
 * TODO: Rely on isEven, but do not highlight bug.
 *
 * @param int $a
 * @return bool
 */
function isOdd($a) {
    return ($a % 2) === 1;
}

/**
 * Arrow functions
 */
$fn1 = fn($x) => $x + $y;

$fn2 = function ($x) use ($y) {
    return $x + $y;
};
