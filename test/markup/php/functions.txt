/**
 * Function declaration
 */
function testMe(string|int $name): int
{
    if (empty($name)) {
        return 0;
    } elseif ($name === 1) {
        return (int) $name;
    }

    switch($name) {
        case '2':
            return 2;
        default:
            throw new \Exception('error');
    }
}
