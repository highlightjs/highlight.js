/**
*
* Simple multi-line comment.
*
* @command test
*
* @switch a  some switch
*
* @param <required>  a required parameter
* @param [optional]  an optional parameter
*
* @global
*
*/
alias test {
  if (!$isid) {
    ; called as a command
    if ($0 >= 1) {
      ; has at least 1 param
      echo -s success, params: $1-
    }
    else {
      ; no params
      echo $color(info) -es * /test: insufficient parameters
      halt
    }
  }
  else {
    /*
    called as an identifier, so must throw an error and halt any
    further commands.
    */
    echo $color(info) -es * test: is not an identifier
    halt
  }
}
