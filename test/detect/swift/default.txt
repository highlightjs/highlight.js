/// A generic stack for equatable items.
public struct Stack<T>: Equatable where T: Equatable {
  private(set) var items: [T]

  public init(items: [T] = []) {
    self.items = items
  }

  public mutating func push(item: T) { /* /* ... */ */ }

  fileprivate func description() -> String {
    if items.isEmpty {
      "The stack is empty"
    } else {
      "\(count) items in the stack"
    }
  }
}
