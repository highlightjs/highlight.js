class Person
  def initialize(@name)
  end

  def greet
    puts "Hi, I'm #{@name}"
  end
end

class Employee < Person
end

employee = Employee.new "John"
employee.greet #=> "Hi, I'm John"
employee.is_a?(Person) #=> true
