\cs_new:Nn \hljs_copy_tmps:
  {
    \tl_set_eq:NN \l_tmpa_tl \@tempa
    \tl_set_eq:NN \l_tmpb_tl \@tempb
    \tl_gset_eq:NN \g_tmpa_tl \@gtempa
    \int_set_eq:NN \l_tmpa_int \@tempcnta
    \int_set_eq:NN \l_tmpb_int \@tempcntb
  }
\def\@copy@temps{%
  \let\@tempa\l_tmpa_tl
  \let\@tempb\l_tmpb_tl
  \global\let\@gtempa\g_tmpa_tl
  \@tempcnta=\l_tmpa_int\relax
  \@tempcntb=\l_tmpb_int\relax
}
